    <script>
        function moveButton() {
            const noBtn = document.getElementById('noBtn');
            
            // Calculate random positions
            // On mobile, we limit the movement to stay mostly within viewable area
            const padding = 50;
            const maxX = window.innerWidth - noBtn.offsetWidth - padding;
            const maxY = window.innerHeight - noBtn.offsetHeight - padding;
            
            const randomX = Math.max(padding, Math.floor(Math.random() * maxX));
            const randomY = Math.max(padding, Math.floor(Math.random() * maxY));
            
            noBtn.style.left = randomX + 'px';
            noBtn.style.top = randomY + 'px';
            noBtn.style.position = 'fixed'; // Switch to fixed so it can move anywhere on screen
        }

        function sayYes() {
            document.getElementById('question-track').classList.add('hidden');
            document.getElementById('answer-track').classList.remove('hidden');
            document.getElementById('noBtn').style.display = 'none';
        }
    </script>
